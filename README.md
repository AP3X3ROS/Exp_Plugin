# Exp_Plugin
This is a Exp plugins that use plugin provided system but not using the Minecraft Exp system.
<pr>
commands:
  glory:
    description: get item by using glory
    usage: /glory <ID> <number>
  setgrade:
    description: Set grade for a player

  giveglory:
    desciption: give glory to a player
    usage: /giveglory <player> <number>
  health:
    description: Set Max health for a player
    usage: /health <player> <number>
  money:
    description: Sign Daily
    usage: /money
  skills:
    description: study skills
    usage: /skills <ID> <number>
  setskills:
    description: set skills for a player
    usage: /setskills <player> <number>
  qdskills:
    description: Sign up for skills [SVIP]
